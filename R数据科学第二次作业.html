#(1)求解一元二次方程的实根
get.root <- function(a,b,c)
{
    #求delta
    delta = b^2 - 4*a*c
    if(delta < 0){
        #无实根
        NA
    } else if(delta == 0){
        #单根
        c(-b/(2*a)) #返回单一向量
    } else{ #两根
        c((-b+sqrt(delta))/(2*a),(-b-sqrt(delta))/(2*a))
    }
}

#判断是否有实根
get.haveRoot <- function(a,b,c)
{
    delta = b^2 - 4*a*c
    if(delta >= 0){
        TRUE
    } else{
        FALSE
    }
}

#判断二次方程有实根概率，n为样本数
get.prob <- function(n)
{
    a <- runif(n,1,5) #1~5均匀分布
    b <- rnorm(n,3,sqrt(10)) #N(3,10)
    c <- rexp(n,rate = 1) #指数分布
    cnt <- 0 #统计有实根数量
    for(i in 1:n){
        if(get.haveRoot(a[i],b[i],c[i]) == TRUE){
            cnt = cnt + 1;
        }
    }
    #是实根概率
    haveRootProbability <- cnt/n
    cat(haveRootProbability)
}

#（7）请随机产生若干直线，验证（5）的结果是最优的
model <- function(c,data){
    c[1] + data$price * c[2]
}

measure_distance <- function(mod, data){
    diff <- data$marketshare - model(mod,data)
    sqrt(mean(diff^2))
}

dist <- NA
CNT = 2500000
a1 = runif(CNT, 0, 1)
a2 = runif(CNT, 0, 1)
for(i in 1:CNT){
  dist[i] <- measure_distance(c(a1[i], a2[i]), hw2_no_na_df)
}
tb <- tibble(
      a1,a2,dist
    )

ggplot(hw2_no_na_df, aes(price, marketshare)) +
geom_point(size = 2, color = "gray20") +
geom_abline(
aes(intercept = a1, slope = a2, color = -dist),
 data = filter(tb, rank(dist) <= 10)
 )
