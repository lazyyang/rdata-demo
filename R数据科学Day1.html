#计算圆周率PI
#算法：测出圆周率pi 在单位正方形中画一单位圆，向单位正方形随机打点，pi=圆形点数/正方形点数 * 4（即面积之比）
CNT <- 10000000
x <- runif(CNT,min = -1,max = 1)  #在-1到1之间随机取100000000个值赋值给x
y <- runif(CNT,min = -1,max = 1)  #在-1到1之间随机取100000000个值赋值给y
a <- sum((x^2+y^2)<1)   #求满足(x^2+y^2)<1的所有点数
b <- CNT   #给b赋值100000000
pi=4*a/b #pi=圆形点数/正方形点数*4（即面积之比）
pi
